api.ts 
import { z } from 'zod';
import { createRouter, createEndpoint, query } from '@xrpc/core';

// Define an endpoint with typed input/output
const greeting = createEndpoint({
  greet: query({
    input: z.object({ name: z.string().min(1) }),  // Validated at runtime
    output: z.object({ message: z.string() }), // Type-safe response
  }),
});

// Export router - generates clients + servers
export const router = createRouter({ greeting });
